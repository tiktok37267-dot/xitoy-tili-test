        /* --- 2. GLOBAL O'ZGARUVCHILAR --- */
        let selectedWords = []; // Tanlab olingan 50 ta so'z
        let currentQuestionIdx = 0;
        let correctAnswersCount = 0;
        let wrongAnswersCount = 0;
        let isAnswered = false; // Savolga javob berildimi?

        /* --- 3. INIT (TIZIMNI ISHGA TUSHIRISH) --- */
        document.addEventListener("DOMContentLoaded", () => {
            // Avvalgi mavzuni tiklash (Dark mode)
            if (localStorage.getItem("theme") === "dark") {
                document.documentElement.setAttribute("data-theme", "dark");
            }
            startNewQuiz();
        });

        /* --- 4. NEW GAME / RESTART FUNKSIYASI --- */
        function startNewQuiz() {
            currentQuestionIdx = 0;
            correctAnswersCount = 0;
            wrongAnswersCount = 0;
            isAnswered = false;

            // 300 ta so'z ichidan ixtiyoriy 50 tasini tanlash
            selectedWords = getRandomSubarray(wordsDatabase, 50);

            // UI elementlarini yashirish/ko'rsatish
            document.getElementById("quiz-screen").style.display = "block";
            document.getElementById("result-screen").style.display = "none";

            // Statistika va birinchi savolni yuklash
            updateStats();
            loadQuestion();
        }

        /* --- 5. SAVOLNI YUKLASH FUNKSIYASI --- */
        function loadQuestion() {
            isAnswered = false;
            document.getElementById("next-button").style.display = "none";

            const currentWord = selectedWords[currentQuestionIdx];

            // UI yangilash
            document.getElementById("question-hanzi").innerText = currentWord.hanzi;
            document.getElementById("question-pinyin").innerText = currentWord.pinyin;

            // Variantlarni yaratish
            generateOptions(currentWord);

            // Progress Barni yangilash
            const progressPercent = (currentQuestionIdx / selectedWords.length) * 100;
            document.getElementById("progress").style.width = `${progressPercent}%`;

            // Avtomatik talaffuz qilish
            speakCurrentWord();
        }

        /* --- 6. VARIANTLARNI GENERATSIYA QILISH --- */
        function generateOptions(correctWord) {
            const optionsBox = document.getElementById("options-box");
            optionsBox.innerHTML = ""; // Tozalash

            // To'g'ri javobdan boshqa barcha so'zlar ro'yxati
            const wrongPool = wordsDatabase.filter(w => w.hanzi !== correctWord.hanzi);
            // 3 ta tasodifiy noto'g'ri javob olish
            const randomWrongWords = getRandomSubarray(wrongPool, 3);

            // To'g'ri javob bilan birlashtirish va aralashtirish
            const optionsList = [correctWord, ...randomWrongWords];
            shuffleArray(optionsList);

            // Tugmalarni ekranga chiqarish
            optionsList.forEach(word => {
                const button = document.createElement("button");
                button.className = "option-btn";
                button.innerText = word.uzbek;
                // Click hodisasi
                button.onclick = () => handleAnswerSelect(button, word, correctWord);
                optionsBox.appendChild(button);
            });
        }

        /* --- 7. JAVOB TANLANGANDA (CLICK) --- */
        function handleAnswerSelect(selectedButton, chosenWord, correctWord) {
            if (isAnswered) return; // Agar javob berilgan bo'lsa, qayta bosilmasin
            isAnswered = true;

            const optionsBox = document.getElementById("options-box");
            const buttons = optionsBox.getElementsByClassName("option-btn");

            // Barcha tugmalarni muzlatish (disabled qilish)
            for (let btn of buttons) {
                btn.setAttribute("disabled", "true");
            }

            // Tekshirish logicasi
            if (chosenWord.hanzi === correctWord.hanzi) {
                // To'g'ri topildi
                selectedButton.classList.add("correct");
                correctAnswersCount++;
            } else {
                // Noto'g'ri topildi
                selectedButton.classList.add("wrong");
                wrongAnswersCount++;

                // To'g'ri javobni yashil qilib ko'rsatish
                for (let btn of buttons) {
                    if (btn.innerText === correctWord.uzbek) {
                        btn.classList.add("correct");
                    }
                }
            }

            // Statistika va keyingi savol tugmasini yangilash
            updateStats();
            document.getElementById("next-button").style.display = "block";
        }

        /* --- 8. KEYINGI SAVOLGA O'TISH --- */
        function nextQuestion() {
            currentQuestionIdx++;
            if (currentQuestionIdx < selectedWords.length) {
                loadQuestion();
            } else {
                showResults();
            }
        }

        /* --- 9. STATISTIKANI YANGILASH --- */
        function updateStats() {
            const left = selectedWords.length - currentQuestionIdx;
            const totalAnswered = correctAnswersCount + wrongAnswersCount;
            const percent = totalAnswered > 0 ? Math.round((correctAnswersCount / totalAnswered) * 100) : 0;

            document.getElementById("stat-correct").innerText = correctAnswersCount;
            document.getElementById("stat-wrong").innerText = wrongAnswersCount;
            document.getElementById("stat-left").innerText = left;
            document.getElementById("stat-percent").innerText = `${percent}%`;
        }

        /* --- 10. NATIJALAR EKRANINI KO'RSATISH --- */
        function showResults() {
            document.getElementById("quiz-screen").style.display = "none";
            document.getElementById("result-screen").style.display = "block";

            const total = selectedWords.length;
            const finalPercent = Math.round((correctAnswersCount / total) * 100);

            document.getElementById("final-score").innerText = `${correctAnswersCount} / ${total}`;
            document.getElementById("final-percent").innerText = `${finalPercent}%`;

            // LocalStorage bilan ishlash
            const lastResultStr = `${correctAnswersCount} ta to'g'ri (${finalPercent}%)`;
            localStorage.setItem("hsk_last_result", lastResultStr);

            // Eng yaxshi rekordni tekshirish va saqlash
            const currentBest = localStorage.getItem("hsk_best_score") || 0;
            if (correctAnswersCount > parseInt(currentBest)) {
                localStorage.setItem("hsk_best_score", correctAnswersCount);
                localStorage.setItem("hsk_best_result", lastResultStr);
            }

            // Ekrandagi yozuvlarni yangilash
            document.getElementById("best-record").innerText = localStorage.getItem("hsk_best_result") || lastResultStr;
            document.getElementById("last-record").innerText = lastResultStr;
        }

        /* --- 11. WEB SPEECH API (TALAFFUZ) --- */
        function speakCurrentWord() {
            if (!('speechSynthesis' in window)) {
                console.warn("Brauzerda ovoz funksiyasi ishlamaydi.");
                return;
            }

            // Hozirgi iyeroglifni olish
            const textToSpeak = selectedWords[currentQuestionIdx].hanzi;
            
            // Oldingi audio jarayonlarni to'xtatish
            window.speechSynthesis.cancel();

            const utterance = new SpeechSynthesisUtterance(textToSpeak);
            utterance.lang = 'zh-CN'; // Xitoy tili (Mandarin)
            utterance.rate = 0.8;     // Biroz sekinroq va aniqroq talaffuz

            window.speechSynthesis.speak(utterance);
        }

        /* --- 12. DARK / LIGHT MODE REJIMINI O'ZGARTIRISH --- */
        function toggleTheme() {
            const currentTheme = document.documentElement.getAttribute("data-theme");
            if (currentTheme === "dark") {
                document.documentElement.removeAttribute("data-theme");
                localStorage.setItem("theme", "light");
            } else {
                document.documentElement.setAttribute("data-theme", "dark");
                localStorage.setItem("theme", "dark");
            }
        }

        /* --- 13. YORDAMCHI ALGORITMLAR (UTILITIES) --- */
        
        // Massiv ichidan tasodifiy 'n' ta element ajratib olish (Fisher-Yates asosida)
        function getRandomSubarray(arr, size) {
            let shuffled = arr.slice(0), i = arr.length, temp, index;
            while (i--) {
                index = Math.floor((i + 1) * Math.random());
                temp = shuffled[index];
                shuffled[index] = shuffled[i];
                shuffled[i] = temp;
            }
            return shuffled.slice(0, size);
        }

        // Massiv elemetlarini aralashtirish (Variantlar uchun)
        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
        }
